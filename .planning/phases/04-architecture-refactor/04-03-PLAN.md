**New screen structure (per D-04):**

```typescript
import { FogWisp } from '@/src/components/scene/FogWisp';
import { Leaf } from '@/src/components/scene/Leaf';
import { useHomeScene } from '@/hooks/useHomeScene';
import { useCheckIn } from '@/hooks/useCheckIn';
import { useQuest } from '@/hooks/useQuest';
import type { HomePhase, Emotion } from '@/src/data/types';
import { loadWorldState, getTodayKey, type WorldState } from '@/lib/storage';
// ... other React/RN imports

export default function HomeScreen() {
  // -- Phase state machine (per D-01, D-04) --
  const [phase, setPhase] = useState<HomePhase>({ phase: 'idle' });
  const [dialogue, setDialogue] = useState('The path is blocked...');
  const [showDialogue, setShowDialogue] = useState(true);
  const [worldState, setWorldState] = useState<WorldState | null>(null);

  const insets = useSafeAreaInsets();
  const { width, height } = useWindowDimensions();
  const sceneConfig = useMemo(() => getSceneConfig(width, height, insets.top), [width, height, insets.top]);

  // -- Load world state on mount (per D-07 — screen owns the load) --
  useEffect(() => {
    async function init() {
      const state = await loadWorldState();
      setWorldState(state);
      // Check if already visited today -> returning phase
      const today = getTodayKey();
      if (state.lastVisitDate === today) {
        setPhase({ phase: 'returning' });
        setDialogue('Welcome back, Lanternkeeper.');
      }
    }
    init();
  }, []);

  // -- Hooks (per D-05) --
  const scene = useHomeScene({ phase, setPhase, setDialogue, setShowDialogue, sceneConfig });
  const checkIn = useCheckIn({ phase, setPhase, setDialogue, setShowDialogue, worldState, setWorldState });
  const quest = useQuest({ phase, setPhase, setDialogue, setShowDialogue, currentQuest: checkIn.currentQuest, newUnlocks: checkIn.newUnlocks, clearNewUnlocks: ... });

  // -- Reset handler --
  const handleReset = useCallback(() => {
    scene.resetScene();
    setPhase({ phase: 'idle' });
    setDialogue('The path is blocked...');
    setShowDialogue(true);
  }, [scene]);

  // ... JSX render tree
}
```

**JSX rewrite — use phase discriminated union for rendering:**

Replace the current boolean-based conditional rendering with phase-based rendering. The ritual panel section currently uses conditions like `readyForCheckIn && !emotion && !alreadyCheckedIn` — replace with `phase.phase === 'check-in'`.

Mapping from current booleans to phases:
- `!pathCleared && !isWalking` (show hints) -> `phase.phase === 'idle' || phase.phase === 'clearing'`
- `readyForCheckIn && !emotion && !alreadyCheckedIn` (show emotion chips) -> `phase.phase === 'check-in'`
- `readyForCheckIn && !emotion && alreadyCheckedIn && !questDone` (show "already checked in") -> `phase.phase === 'returning'`
- `emotion && !questDone` (show quest) -> `phase.phase === 'quest'`
- `questDone` (show rest/walk again) -> `phase.phase === 'feedback' || phase.phase === 'complete'`
- `!readyForCheckIn && !isWalking` (show waiting) -> `phase.phase === 'idle' || phase.phase === 'clearing'` (when not yet started clearing, show "Clear the path to continue")

For rendering with phase data: when `phase.phase === 'quest'`, access `phase.quest` for the quest text and `phase.emotion` for the selected emotion (per D-03).
